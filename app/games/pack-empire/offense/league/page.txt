'use client';
// nfl-minigames-hub/app/games/pack-empire/offense/league/page.tsx

import { useState, useEffect } from 'react';
import Link from 'next/link';
import { createClient } from '@/lib/supabase/client';
import { getMyLeagues, getLeagueByJoinCode, joinLeague, getMyMembership, getLeagueMembers } from '@/lib/league/db';
import type { DBLeague } from '@/lib/league/db';

export default function LeagueHubPage() {
  const supabase = createClient();
  const [userId, setUserId]       = useState<string | null>(null);
  const [leagues, setLeagues]     = useState<DBLeague[]>([]);
  const [joinCode, setJoinCode]   = useState('');
  const [joinError, setJoinError] = useState('');
  const [joining, setJoining]     = useState(false);
  const [loading, setLoading]     = useState(true);

  useEffect(() => {
    (async () => {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) { setLoading(false); return; }
      setUserId(user.id);
      const my = await getMyLeagues(user.id);
      setLeagues(my);
      setLoading(false);
    })();
  }, []);

  async function handleJoin() {
    if (!userId || !joinCode.trim()) return;
    setJoining(true);
    setJoinError('');
    try {
      const league = await getLeagueByJoinCode(joinCode.trim());
      if (!league) { setJoinError('League not found. Check your code.'); return; }
      if (league.phase !== 'pregame') { setJoinError('This league has already started.'); return; }
      const existing = await getMyMembership(league.id, userId);
      if (existing) { window.location.href = `/games/pack-empire/offense/league/${league.id}`; return; }
      const members = await getLeagueMembers(league.id);
      if (members.length >= league.max_players) { setJoinError('League is full.'); return; }
      await joinLeague(league.id, userId);
      window.location.href = `/games/pack-empire/offense/league/${league.id}`;
    } catch {
      setJoinError('Something went wrong. Try again.');
    } finally {
      setJoining(false);
    }
  }

  const phaseLabel: Record<string, string> = {
    pregame: 'Waiting to start', regular: 'Regular season',
    gauntlet: 'Playoffs — gauntlet', finals: 'Playoffs — finals', complete: 'Complete',
  };

  return (
    <>
      <div className="lhub-root">

        {/* ── Hero ── */}
        <div className="lhub-hero">
          <div className="lhub-hero-bg">
            <div className="lhub-grid" />
            <div className="lhub-flare-1" />
            <div className="lhub-flare-2" />
            <div className="lhub-scanline" />
          </div>
          <div className="lhub-hero-inner">
            <Link href="/games/pack-empire/offense" className="lhub-back-pe">
  ← Pack Empire
</Link>
            <div className="lhub-eyebrow">
              <span className="lhub-dot" /><span className="lhub-dot" /><span className="lhub-dot" />
              &nbsp;PACK EMPIRE&nbsp;
              <span className="lhub-dot" /><span className="lhub-dot" /><span className="lhub-dot" />
            </div>
            <h1 className="lhub-title">
              <span className="lhub-t1">LEAGUE</span>
              <span className="lhub-t2">MODE</span>
            </h1>
            <p className="lhub-tagline">Draft. Lock. Dominate. Eight weeks to build a dynasty.</p>
            <div className="lhub-stats">
              <div className="lhub-stat"><span className="lhub-sn">8</span><span className="lhub-sl">WEEKS</span></div>
              <div className="lhub-sdiv" />
              <div className="lhub-stat"><span className="lhub-sn">11</span><span className="lhub-sl">PLAYERS</span></div>
              <div className="lhub-sdiv" />
              <div className="lhub-stat"><span className="lhub-sn">12</span><span className="lhub-sl">MAX TEAMS</span></div>
            </div>
          </div>
        </div>

        {/* ── How it works ── */}
        <div className="lhub-how-outer"><div className="lhub-how">
          {[
            { icon: '📦', n: '01', title: 'Draft weekly',      body: 'Open packs every week. Score updates as you lock in better players.' },
            { icon: '🔒', n: '02', title: 'Lock one per week', body: 'After each draft, keep one player permanently. 8 weeks = 8 locks.' },
            { icon: '⚡', n: '03', title: 'Chemistry matters', body: 'Stack teammates for synergy bonuses. Mix rivals and pay the penalty.' },
            { icon: '🏆', n: '04', title: 'Win the league',    body: 'Top half make playoffs. Survive the gauntlet. Win the best-of-3.' },
          ].map(c => (
            <div key={c.n} className="lhub-how-card">
              <div className="lhub-how-n">{c.n}</div>
              <div className="lhub-how-icon">{c.icon}</div>
              <div className="lhub-how-title">{c.title}</div>
              <div className="lhub-how-body">{c.body}</div>
            </div>
          ))}
        </div></div>

        {/* ── Action panel ── */}
        <div className="lhub-panel">
          {loading ? (
            <div className="lhub-loading"><div className="lhub-spinner" /></div>
          ) : !userId ? (
            <div className="lhub-gated">
              <div className="lhub-gated-icon">🔐</div>
              <div className="lhub-gated-title">SIGN IN TO COMPETE</div>
              <div className="lhub-gated-sub">League Mode requires an account.</div>
              <Link href="/login" className="lhub-cta">SIGN IN →</Link>
            </div>
          ) : (
            <>
              <div className="lhub-actions">
                <Link href="/games/pack-empire/offense/league/create" className="lhub-create">
                  <span className="lhub-create-plus">+</span>
                  <span>
                    <span className="lhub-create-title">CREATE LEAGUE</span>
                    <span className="lhub-create-sub">Set up your own league as commissioner</span>
                  </span>
                  <span className="lhub-create-arrow">›</span>
                </Link>

                <div className="lhub-join">
                  <div className="lhub-join-label">JOIN WITH CODE</div>
                  <div className="lhub-join-row">
                    <input
                      className="lhub-code-input"
                      placeholder="ENTER CODE…"
                      value={joinCode}
                      onChange={e => setJoinCode(e.target.value.toUpperCase())}
                      maxLength={10}
                      onKeyDown={e => e.key === 'Enter' && handleJoin()}
                    />
                    <button className="lhub-join-btn" onClick={handleJoin} disabled={joining || !joinCode.trim()}>
                      {joining ? '…' : 'JOIN'}
                    </button>
                  </div>
                  {joinError && <div className="lhub-join-err">{joinError}</div>}
                </div>
              </div>

              {leagues.length > 0 ? (
                <div className="lhub-my">
                  <div className="lhub-my-label">
                    <span className="lhub-my-line" />MY LEAGUES<span className="lhub-my-line" />
                  </div>
                  {leagues.map(lg => (
                    <Link key={lg.id} href={`/games/pack-empire/offense/league/${lg.id}`} className="lhub-row">
                      <div className="lhub-row-bar" />
                      <div className="lhub-row-body">
                        <div className="lhub-row-name">{lg.name}</div>
                        <div className="lhub-row-meta">
                          {phaseLabel[lg.phase] ?? lg.phase}
                          {lg.phase === 'regular' && ` · Week ${lg.current_week} of 8`}
                        </div>
                      </div>
                      <div className="lhub-row-code">{lg.id}</div>
                      <div className="lhub-row-arrow">›</div>
                    </Link>
                  ))}
                </div>
              ) : (
                <div className="lhub-empty">
                  <span className="lhub-empty-icon">🏟️</span>
                  <span className="lhub-empty-msg">No leagues yet — create one or enter a code above.</span>
                </div>
              )}
            </>
          )}
        </div>
      </div>

      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Rajdhani:wght@400;500;600;700&family=Orbitron:wght@700;800;900&display=swap');

        /* ── palette ──────────────────────────────────────────────────────
           #ff4500  vivid red-orange  (primary actions, accents)

           #fff0e8  near-white        (headings, important values)
        ───────────────────────────────────────────────────────────────── */

        *{box-sizing:border-box}
        .lhub-root{font-family:'Rajdhani',sans-serif;color:#ffd4a8;background:#1c0a00;min-height:100vh}

        /* Hero */
        .lhub-hero{position:relative;overflow:hidden;padding:0}
        .lhub-hero-bg{position:absolute;inset:0;background:linear-gradient(180deg,#2e1200 0%,#1c0a00 100%)}
        .lhub-grid{position:absolute;inset:0;background-image:linear-gradient(rgba(255,69,0,.15) 1px,transparent 1px),linear-gradient(90deg,rgba(255,69,0,.15) 1px,transparent 1px);background-size:44px 44px;mask-image:radial-gradient(ellipse 80% 70% at 50% 0%,black 30%,transparent 100%)}
        .lhub-flare-1{position:absolute;top:-80px;left:50%;transform:translateX(-50%);width:600px;height:400px;background:radial-gradient(ellipse,rgba(255,69,0,.28) 0%,transparent 65%);pointer-events:none;animation:lhub-pulse 4s ease-in-out infinite}
        .lhub-flare-2{position:absolute;bottom:0;right:8%;width:280px;height:280px;background:radial-gradient(ellipse,rgba(255,140,0,.14) 0%,transparent 70%);pointer-events:none;animation:lhub-pulse 6s ease-in-out infinite reverse}
        @keyframes lhub-pulse{0%,100%{opacity:.7}50%{opacity:1}}
        .lhub-scanline{position:absolute;inset:0;background:repeating-linear-gradient(0deg,transparent,transparent 3px,rgba(0,0,0,.1) 3px,rgba(0,0,0,.1) 4px);pointer-events:none;opacity:.35}

        .lhub-back-pe{color:#7a3a10;font-size:.82rem;letter-spacing:.1em;font-weight:600;text-decoration:none;transition:.15s;display:block;padding:.8rem 2rem;border-bottom:1px solid #2e1200}
        .lhub-back-pe:hover{color:#ff8c00}

        .lhub-hero-inner{position:relative;z-index:2;text-align:center;padding:3rem 2rem 2.5rem;max-width:900px;margin:0 auto}
        .lhub-eyebrow{display:flex;align-items:center;justify-content:center;gap:.35rem;font-family:'Orbitron',sans-serif;font-size:.6rem;letter-spacing:.28em;color:#7a3a10;margin-bottom:.8rem}
        .lhub-dot{width:5px;height:5px;border-radius:50%;background:#ff4500;flex-shrink:0;box-shadow:0 0 4px #ff4500}

        .lhub-title{margin:0 0 .7rem;line-height:.88}
        .lhub-t1{display:block;font-family:'Orbitron',sans-serif;font-size:clamp(3.5rem,12vw,6.5rem);font-weight:900;color:#fff0e8;letter-spacing:-.02em;text-shadow:0 0 40px rgba(255,69,0,.3)}
        .lhub-t2{display:block;font-family:'Orbitron',sans-serif;font-size:clamp(3.5rem,12vw,6.5rem);font-weight:900;background:linear-gradient(135deg,#ff4500 0%,#ff6a00 45%,#ff8c00 100%);-webkit-background-clip:text;-webkit-text-fill-color:transparent;background-clip:text;letter-spacing:-.02em;filter:drop-shadow(0 0 24px rgba(255,69,0,.55))}

        .lhub-tagline{font-size:clamp(.9rem,2.5vw,1.1rem);color:#a05030;font-weight:500;letter-spacing:.03em;margin-bottom:1.6rem}

        .lhub-stat{display:flex;flex-direction:column;align-items:center;padding:0 1.8rem}
        .lhub-sn{font-family:'Orbitron',sans-serif;font-size:2rem;font-weight:900;color:#ff8c00;line-height:1;text-shadow:0 0 16px rgba(255,140,0,.4)}
        .lhub-sl{font-size:.58rem;letter-spacing:.22em;color:#7a3a10;margin-top:.2rem}
        .lhub-sdiv{width:1px;height:44px;background:linear-gradient(180deg,transparent,#4a2000,transparent)}

        /* How it works */
        .lhub-how-outer{border-top:1px solid #4a2000;border-bottom:1px solid #4a2000;background:#2e1200}.lhub-how{display:grid;grid-template-columns:repeat(4,1fr);gap:1px;background:#2e1200;max-width:80%;margin:0 auto}
        .lhub-how-card{background:#1c0a00;padding:1.2rem 1rem;position:relative;overflow:hidden;transition:.15s}
        .lhub-how-card:hover{background:#220e00}
        .lhub-how-card::before{content:'';position:absolute;top:0;left:0;right:0;height:2px;background:linear-gradient(90deg,transparent,rgba(255,69,0,.5),transparent)}
        .lhub-how-n{font-family:'Orbitron',sans-serif;font-size:.55rem;letter-spacing:.2em;color:#4a2000;position:absolute;top:.6rem;right:.7rem}
        .lhub-how-icon{font-size:1.5rem;margin-bottom:.4rem}
        .lhub-how-title{font-family:'Rajdhani',sans-serif;font-weight:700;font-size:.95rem;color:#fff0e8;letter-spacing:.06em;margin-bottom:.3rem;text-transform:uppercase}
        .lhub-how-body{font-size:.8rem;color:#7a3a10;line-height:1.5;font-weight:500}

        /* Panel */
        .lhub-panel{max-width:900px;margin:0 auto;padding:1.8rem 2rem 4rem}

        .lhub-loading{display:flex;justify-content:center;padding:3rem}
        .lhub-spinner{width:32px;height:32px;border:2px solid #4a2000;border-top-color:#ff4500;border-radius:50%;animation:lhub-spin .8s linear infinite}
        @keyframes lhub-spin{to{transform:rotate(360deg)}}

        .lhub-gated{text-align:center;padding:3rem 1rem;background:#2e1200;border:1px solid #4a2000;border-radius:14px}
        .lhub-gated-icon{font-size:2.5rem;margin-bottom:.6rem}
        .lhub-gated-title{font-family:'Orbitron',sans-serif;font-size:.9rem;color:#ff4500;letter-spacing:.14em;margin-bottom:.4rem}
        .lhub-gated-sub{color:#7a3a10;font-size:.9rem;margin-bottom:1.4rem}
        .lhub-cta{display:inline-block;background:linear-gradient(135deg,#cc2200,#ff4500,#ff8c00);color:#fff0e8;border-radius:9px;padding:.75rem 2rem;font-family:'Rajdhani',sans-serif;font-weight:700;font-size:1rem;letter-spacing:.14em;text-decoration:none;transition:.2s}
        .lhub-cta:hover{filter:brightness(1.1);transform:scale(1.02)}

        /* Actions row */
        .lhub-actions{display:grid;grid-template-columns:1fr 1fr;gap:1rem;margin-bottom:1.6rem}

        .lhub-create{display:flex;align-items:center;gap:1rem;background:#2e1200;border:1.5px solid #ff4500;border-radius:12px;padding:1.2rem 1.4rem;text-decoration:none;transition:.2s;position:relative;overflow:hidden}
        .lhub-create::before{content:'';position:absolute;inset:0;background:linear-gradient(135deg,rgba(255,69,0,.08),transparent);pointer-events:none}
        .lhub-create:hover{background:#3a1800;border-color:#ff8c00;box-shadow:0 0 20px rgba(255,69,0,.2)}
        .lhub-create-plus{font-family:'Orbitron',sans-serif;font-size:2.2rem;font-weight:900;color:#ff4500;line-height:1;flex-shrink:0;text-shadow:0 0 12px rgba(255,69,0,.5)}
        .lhub-create-title{display:block;font-family:'Orbitron',sans-serif;font-size:.8rem;font-weight:800;color:#ff8c00;letter-spacing:.1em;margin-bottom:.2rem}
        .lhub-create-sub{display:block;font-size:.78rem;color:#7a3a10;font-weight:500}
        .lhub-create-arrow{margin-left:auto;color:#4a2000;font-size:1.5rem;transition:.15s}
        .lhub-create:hover .lhub-create-arrow{color:#ff4500}

        .lhub-join{background:#2e1200;border:1px solid #4a2000;border-radius:12px;padding:1.2rem 1.4rem}

        .lhub-join-row{display:flex;gap:.5rem}
        .lhub-code-input{flex:1;background:#1c0a00;border:1.5px solid #4a2000;border-radius:7px;padding:.65rem .9rem;font-family:'Orbitron',sans-serif;font-size:.82rem;color:#fff0e8;outline:none;letter-spacing:.14em;transition:.15s;min-width:0}
